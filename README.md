## Install

```bash
npm install --save crh-app
```

## Usage

```jsx
import React, { Component } from 'react'

import { useMyHook } from 'crh-app'

const Example = () => {
  const example = useMyHook()
  return (
    <div>{example}</div>
  )
}
```
